	public class Controler {

        private SldWorks swApp;
        private ModelDoc2 model;
        private Service service;
        private Weldment wd;
        private Body2 baseBody;
        public static clsPropMgr pm;
        private string macroPath;


        public Controler(SldWorks sw)
        {
            swApp = sw;
            model = (ModelDoc2)swApp.ActiveDoc;

            if (model == null || model.GetType() != (int)swDocumentTypes_e.swDocPART)
            {
                swApp.SendMsgToUser("Откройте деталь");
                return;
            }

            macroPath = swApp.GetCurrentMacroPathName();

            service = new Service(swApp);

            baseBody = service.GetBaseBody();   // 🔴 FIX

            wd = service.GetWeldment();
            if (wd == null) return;

            pm = new clsPropMgr(swApp, service, wd);

           // pm.action += OnPreview;
            pm.ok += OnCommit;
            pm.angle += OnAngleChanged;
            pm.rotate += OnRotateChanged;

            pm.Show();

            while (pm != null && pm.IsOpen)
                Application.DoEvents();

           GC.KeepAlive(pm);
        }

 
        // 🔷 PREVIEW
    
        private void OnPreview()
        {
            try
            {
                service.BuildCutBody(wd);   // без baseBody
                service.UpdatePreview();
            }
            catch (Exception ex)
            {
                swApp.SendMsgToUser(ex.Message);
            }
        }

     
  
        private void OnCommit()
        {
            try
            {
                InsertMacroFeature();
            }
            catch (Exception ex)
            {
                swApp.SendMsgToUser(ex.Message);
            }
        }

        private void InsertMacroFeature()
        {
            string[] paramNames = new string[]
            {
                "Angle",
                "Offset"
            };

            int[] paramTypes = new int[]
            {
                (int)swMacroFeatureParamType_e.swMacroFeatureParamTypeDouble,
                (int)swMacroFeatureParamType_e.swMacroFeatureParamTypeDouble
            };

            string[] paramValues = new string[]
            {
                wd.axis.ToString(),
                wd.s.ToString()
            };

            string[] methods = new string[]
            {
                macroPath, "Module1", "swmMain",
                macroPath, "Module1", "swmPM",
                "", "", ""
            };

            object[] bodies = new object[] { baseBody };

            model.FeatureManager.InsertMacroFeature3(
                "AngleSurf",
                "",
                methods,
                paramNames,
                paramTypes,
                paramValues,
                null,
                null,
                bodies,
                null,
                (int)swMacroFeatureOptions_e.swMacroFeatureByDefault);
        }




        private void OnAngleChanged(double value)
        {
            wd.axis = value;
        }

        private void OnRotateChanged(double value)
        {
            wd.rotationAxis = value;
        }


        public void Completion(){

		}

		public void Reverse(){

		}

		public void Rotate(){

		}

	}

    public class Service
    {

        private FeatureManager fm;
        private MathUtility mathUtility;
        private SelectionMgr slMg;
        public SldWorks swApp;
        private ModelDoc2 model;
        private PartDoc part;
        private SelectionMgr skMgr;
        private ModelDocExtension ext;
        SketchManager swSkMgr = default(SketchManager);
        bool statusSelect;

        private Body2 previewBody;  // Preview body
        private Body2 toolBody;     // Tool body
        private Body2 baseBody;     // Original body


        public Service(SldWorks swSld)
        {
            swApp = swSld;
            model = (ModelDoc2)swApp.ActiveDoc;
            if (model == null || model.GetType() != (int)swDocumentTypes_e.swDocPART)
            {
                swApp.SendMsgToUser("Откройте деталь перед запуском макроса.");
                return;
            }
            CheckingIsPart();
            part = (PartDoc)model;
            mathUtility = (MathUtility)swApp.GetMathUtility();
            ext = model.Extension;
            fm = model.FeatureManager;
            skMgr = (SelectionMgr)model.SelectionManager;
            swSkMgr = model.SketchManager;
            baseBody = GetBaseBody();
        }

        public Body2 BuildCutBody(Weldment wd)
        {
            model.ClearSelection2(true);

            double[] box = GetPointsBoundinBox();
            double[][] axisPts = wd.GetCentetPonts(box);

            string axis = AddAxis(axisPts);
            string plane = AddPlaneWork(axis, wd.namePlane, (int)wd.rotationAxis);

            // --- первый профиль ---
            double[] p1 = wd.GetPointsFirstExtrude(true);
            string sk1 = CreateSketch(plane, p1);

            // --- второй профиль ---
            double[] p2 = wd.GetPointsSecondExtrude(p1);
            string sk2 = CreateSketch(plane, p2);

            // --- строим тело инструмента ---
            Body2 toolBody = CreateToolBody(new string[] { sk1 });

            return toolBody;
        }
        public bool PreviewCutBody(Weldment wd, Body2 originalBody)
        {
            // --- 1. Клонируем тело, чтобы не трогать оригинал ---
            Body2 previewBody = (Body2)originalBody.Copy();

            // --- 2. Строим тело инструмента (инструмент для выреза) ---
            double[] box = GetPointsBoundinBox();
            double[][] axisPts = wd.GetCentetPonts(box);

            string axisName = AddAxis(axisPts);
            string planeName = AddPlaneWork(axisName, wd.namePlane, (int)wd.rotationAxis);

            // Первый профиль
            double[] p1 = wd.GetPointsFirstExtrude(true);
            string sk1 = CreateSketch(planeName, p1);

            // Второй профиль
            double[] p2 = wd.GetPointsSecondExtrude(p1);
            string sk2 = CreateSketch(planeName, p2);

            // Создаем тело инструмента (tool body)
            Body2 toolBody = CreateToolBody(new string[] { sk1, sk2 });
            if (toolBody == null) return false;

            // --- 3. Выполняем булеву операцию Cut ---
            int err;
            object ok = previewBody.Operations2(
                (int)swBodyOperationType_e.SWBODYCUT, // Cut
                (object)toolBody,
                out err
            );

            if ( err != 0)
            {
                swApp.SendMsgToUser("Не удалось выполнить булеву операцию. Код ошибки: " + err);
                return false;
            }

            // --- 4. Показать временно для preview ---
            previewBody.Display3(model, 0, 0);

            return true;
        }

        // ?? SKETCH

        public string CreateSketch(string planeName, double[] pts)
        {
            model.ClearSelection2(true);

            ext.SelectByID2(planeName, "PLANE", 0, 0, 0, false, 0, null, 0);

            swSkMgr.InsertSketch(false);

            CreateClosedSketch(pts);

            Sketch sk = swSkMgr.ActiveSketch as Sketch;
            Feature feat = (Feature)sk;

            string name = feat.Name;

            // фиксируем
            FullyDefineSketch(name);

            swSkMgr.InsertSketch(false);

            return name;
        }

        private void CreateClosedSketch(double[] pts)
        {
            swSkMgr.CreateLine(
                pts[0], pts[1], pts[2],
                pts[6], pts[7], pts[8]);

            swSkMgr.CreateLine(
                pts[3], pts[4], pts[5],
                pts[6], pts[7], pts[8]);

            swSkMgr.CreateLine(
                pts[0], pts[1], pts[2],
                pts[3], pts[4], pts[5]);
        }

        private void FullyDefineSketch(string name)
        {
            model.ClearSelection2(true);

            ext.SelectByID2(name, "SKETCH", 0, 0, 0, false, 0, null, 0);
            model.SketchAddConstraints("sgFIXED");
        }


        // ?? TOOL BODY (через временный extrude)

        private Body2 CreateToolBody(string[] sketches)
        {
            model.ClearSelection2(true);

            foreach (var sk in sketches)
                ext.SelectByID2(sk, "SKETCH", 0, 0, 0, true, 0, null, 0);

            Feature feat = fm.FeatureExtrusion3(
                false,   // solid
                false,   //Flip
                false,   //Dir
                (int)swEndConditions_e.swEndCondBlind,
                (int)swEndConditions_e.swEndCondBlind,
                0.01, 0.01,
                false,  //Dchk1
                false,  ////Dchk2
                false,
                false, 
                0.0,
                0.0,
                false,     //OffsetReverse1            
                false,     //OffsetReverse2
                false,     //TranslateSurface1
                true,      //TranslateSurface2  
                true,      //Merge 
                true,      //UseFeatScope
                true,      //UseAutoSelect 
                0,        //T0
                0.0,      //StartOffset
                false );   //FlipStartOffset  

            if (feat == null) return null;
            
            // Через PartDoc
            PartDoc part = (PartDoc)model;
            object[] bodies = (object[])part.GetBodies2((int)swBodyType_e.swSolidBody, true);

            if (bodies == null || bodies.Length == 0)
                throw new Exception("Нет solid body");

            Body2 body = bodies[bodies.Length - 1] as Body2; // Последнее тело = инструмент
            toolBody = body;

            return body;
        }


        public void UpdatePreview()
        {
            if (baseBody == null || toolBody == null)
                return;

            // Удаляем старый preview
            if (previewBody != null)
                previewBody.HideBody(true);

            // Копируем базовое тело
            previewBody = baseBody.Copy() as Body2;

            // Булев вырез
            int err;
            int swBodyOperationType = (int)swBodyOperationType_e.SWBODYCUT; // Вырез

            object ok = previewBody.Operations2(
                swBodyOperationType, // Cut
                (object)toolBody,    // Тело инструмента
                out err
            );

            if ( err != 0)
                throw new Exception("Булева операция не удалась, код ошибки: " + err);

            // Отображаем preview
            previewBody.Display3(model, 0, 0); // 0 = временно
        }

        public string AddAxis(double[][] pts)
        {
            double[] A = pts[0];
            double[] B = pts[1];

            swSkMgr.Insert3DSketch(true);
            var seg = swSkMgr.CreateLine(A[0], A[1], A[2], B[0], B[1], B[2]);
            seg.ConstructionGeometry = true;
            swSkMgr.Insert3DSketch(false);

            model.ClearSelection2(true);
            seg.Select(true);

            model.InsertAxis2(true);

            Feature feat = (Feature)skMgr.GetSelectedObject6(1, 0);
            return feat.Name;
        }


        // ?? PLANE

        public string AddPlaneWork(string axisName, string basePlane, int angle)
        {
            model.ClearSelection2(true);

            ext.SelectByID2(axisName, "AXIS", 0, 0, 0, false, 0, null, 0);

        public double[] GetPointsBoundinBox()

        public Body2 GetBaseBody()
        {
            // Получаем все solid тела детали
            object[] bodies = (object[])part.GetBodies2((int)swBodyType_e.swSolidBody, true);

            if (bodies == null || bodies.Length == 0)
                throw new Exception("Нет solid body в детали");

            Body2 body = bodies[0] as Body2;
            if (body == null)
                throw new Exception("Не удалось привести тело к Body2");

            return body;
        }


        public Weldment GetWeldment()
        {
            Feature swFeat = (Feature)model.FirstFeature();
            StructuralMemberFeatureData weldData = null;
            while (swFeat != null)
            {
                string tmp = swFeat.GetTypeName();
                if (swFeat.GetTypeName() == "WeldMemberFeat")
                {

                    weldData = (StructuralMemberFeatureData)swFeat.GetDefinition();
                    break;
                }
                else
                {
                    swFeat = (Feature)swFeat.GetNextFeature();
                }

            }

            if (weldData == null)
            {
                swApp.SendMsgToUser("No Structural Member (элемент сварной балки).");
                return null;
            }

          
            // Получаем группы и сегменты и проверяем что ровно одна группа и ровно один сегмент
            weldData.AccessSelections(model, null);
            object[] groups = (object[])weldData.Groups;

            if (groups == null || groups.Length != 1)
            {
                swApp.SendMsgToUser("Элемент должен содержать ровно одну группу.");
                weldData.ReleaseSelectionAccess();
                return null;
            }

            StructuralMemberGroup group = (StructuralMemberGroup)groups[0];
            object[] segs = (object[])group.Segments;

            if (segs == null || segs.Length != 1)
            {
                swApp.SendMsgToUser("В группе должен быть ровно один сегмент.");
                weldData.ReleaseSelectionAccess();
                return null;
            }


            // Получаем сегмент и линию
            SketchSegment seg = (SketchSegment)segs[0];
            SketchLine line = seg as SketchLine;
            if (line == null)
            {
                swApp.SendMsgToUser("Сегмент должен быть линией.");
                weldData.ReleaseSelectionAccess();
                return null;
            }
            // --- 3) Получаем плоскость эскиза, на которой лежит исходный сегмент ---
            Sketch sketch = (Sketch)seg.GetSketch();
            if (sketch == null)
            {
                swApp.SendMsgToUser("Не удалось получить эскиз сегмента.");
                return null;
            }
            int nEntType=0;
            Feature pPlaneFeat = null;
            Entity pEntity =(Entity)sketch.GetReferenceEntity(ref nEntType);
            if (nEntType == (int)swSelectType_e.swSelDATUMPLANES) {
                pPlaneFeat = (Feature)pEntity;
            }

            string namePlane = pPlaneFeat.Name;

            // Получаем начало и конец линии (координаты в модели)
            ISketchPoint point1 = (ISketchPoint)line.GetStartPoint2(); // {x,y,z}
            ISketchPoint point2 = (ISketchPoint)line.GetEndPoint2();
            double[] p1 = new double[3];
            double[] p2 = new double[3];
            p1[0] = point1.X;
            p1[1] = point1.Y;
            p1[2] = point1.Z;
            p2[0] = point2.X;
            p2[1] = point2.Y;
            p2[2] = point2.Z;        //To global ..............
            // Сохраняем угол (если понадобится)
            double angle = group.Angle;
            weldData.ReleaseSelectionAccess();
            string templatePath = weldData.WeldmentProfilePath;
            string template = Path.GetFileNameWithoutExtension(templatePath);
            string[] parametry = template.Split('x');
            Weldment profile = new Weldment(template);
            profile.SetCenterPoits(p1, p2);
            profile.namePlane = namePlane;
            return profile;
        }
